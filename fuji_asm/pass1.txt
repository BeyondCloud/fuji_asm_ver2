getDgt	PROC	
MOV	DL,	'?'	
CALL	putChr	
MOV	AH,	01H	
INT	21H	
CMP	AL,	'0'	
JLE	gError	
CMP	AL,	'9'	
JG	gError	
MOV	DL,	AL	
SUB	AL,	48	
RET	
MOV	DL,	BEL	
CALL	putChr	
JMP	gLoop	
getDgt	ENDP	
putChr	PROC	
CMP	DL,	0	
JE	NoMore	
MOV	AH,	02H	
INT	21H	
RET	
putChr	ENDP	
NewLine	PROC	
MOV	DL,	CR	
CALL	putChr	
MOV	DL,	LF	
CALL	putChr	
RET	
NewLine	ENDP	
doComp	PROC	
MOV	AX,	0	
MOV	DH,	0	
CMP	DX,	0	
JLE	doDone	
ADD	AX,	DX	
DEC	DX	
JMP	doLoop	
DB	0D4H	
DB	00AH	
ADD	AX,	3030H	
RET	
doComp	ENDP	
MAIN	PROC	
MOV	AX,	@DATA	
MOV	DS,	AX	
LEA	DX,	Start1	
MOV	AH,	09H	
INT	21H	
CALL	NewLine	
Call	getDgt	
MOV	uIdgt,	DL	
MOV	DL,	AL	
CALL	doComp	
XCHG	AH,	AL	
MOV	uRval,	AX	
LEA	DX,	uMSG	
MOV	AH,	09H	
INT	21H	
CALL	NewLine	
LEA	DX,	Goodbye	
MOV	AH,	09H	
INT	21H	
MOV	AX,4C00H	
INT	21H	
MAIN	ENDP	
END
